def getnum(a,b):
  c=0
  count=0
  while a and b:
    if a%10+b%10+c>9:c=1
    else:c=0
    count+=c
    a,b=a//10,b//10
  while c and a:
    if a%10+c>9:c=1
    else: c=0
    count+=c
    a=a//10
  while c and b:
    if b%10+c>9:c=1
    else: c=0
    count+=c
    b=b//10
  return count
